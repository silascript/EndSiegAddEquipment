# 零级装甲车
country_event = {
	id = endaddeq.event.armoredcar.902000
	title = endaddeq.event.armoredcar.902000.t
	desc = endaddeq.event.armoredcar.902000.d

	is_triggered_only = yes
	trigger = {
		is_ai = no
	}

	#移除idea
	immediate = {
		hidden_effect = { 
			remove_ideas = eaeq_armored_car_0 
		}
	}

	# 添加5000装甲车
	option = {
		name = endaddeq.event.armoredcar.902000.a  

		add_equipment_to_stockpile = {
			type = armored_car_equipment_0 
			amount = 5000
		}
	}

	# 添加1w装甲车
	option = {
		name = endaddeq.event.armoredcar.902000.b  

		add_equipment_to_stockpile = {
			type = armored_car_equipment_0 
			amount = 10000
		}
	}

	# 添加5w装甲车
	option = {
		name = endaddeq.event.armoredcar.902000.c  

		add_equipment_to_stockpile = {
			type = armored_car_equipment_0 
			amount = 50000
		}
	}


	#关闭
	option = {
		name = endaddeq.event.common.b
	}


} # 零级装甲车
